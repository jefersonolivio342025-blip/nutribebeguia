document.addEventListener('click', function(e) { var target = e.target.closest('a') || e.target.closest('button'); if (target) { // Busca o atributo href de forma segura var href = target.getAttribute('href') || ""; // Verifica se e um link de checkout (Kiwify ou Hotmart) var isCheckoutBtn = href.indexOf('kiwify') !== -1 || href.indexOf('hotmart') !== -1 || target.classList.contains('btn-compra'); if (isCheckoutBtn) { // 1. Registra o clique no seu Dashboard do Supabase if (typeof trackNutri === 'function') { trackNutri('clique', { destino: 'checkout_externo' }); } // 2. Dispara o evento de InitiateCheckout no Facebook if (typeof fbq === 'function') { fbq('track', 'InitiateCheckout'); } // 3. Tenta anexar as UTMs se for um link externo (http) if (href.indexOf('http') === 0) { try { var p = getParams(); var url = new URL(href); url.searchParams.set('utm_source', p.source); url.searchParams.set('utm_content', p.content); target.href = url.toString(); } catch (err) { console.warn("Nao foi possivel anexar UTMs ao link."); } } } } });
