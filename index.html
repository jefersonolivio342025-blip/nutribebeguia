document.addEventListener('click', function(e) { var target = e.target.closest('a') || e.target.closest('button'); if (target) { var href = target.getAttribute('href') || ""; var isCheckoutBtn = href.indexOf('kiwify') !== -1 || href.indexOf('hotmart') !== -1 || target.classList.contains('btn-compra'); if (isCheckoutBtn) { // Impede o redirecionamento imediato para dar tempo ao rastreio e.preventDefault(); // 1. Registra no seu Dashboard if (typeof trackNutri === 'function') { trackNutri('clique', { destino: 'checkout_externo' }); } // 2. Dispara o Facebook if (typeof fbq === 'function') { fbq('track', 'InitiateCheckout'); } // 3. Prepara o link com UTMs var finalUrl = href; if (href.indexOf('http') === 0) { try { var p = getParams(); var url = new URL(href); url.searchParams.set('utm_source', p.source); url.searchParams.set('utm_content', p.content); finalUrl = url.toString(); } catch (err) { console.warn("Erro nas UTMs"); } } // Espera 500ms para garantir o envio e entao redireciona setTimeout(function() { window.location.href = finalUrl; }, 500); } } });
